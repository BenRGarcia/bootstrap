Other things:

  1. Terminal Configuration
    • vim ~/.bash_profile
    • Paste the following into the .bash_profile file 
      #The below statements will print to the terminal upon opening bash window
echo "Welcome, SeeBenProgram!"
echo "Personal settings have been loaded from ~/.bash_profile"
echo "Happy Coding!"
echo "======================================================="

#The below command changes the terminal prompt environment
orange=$(tput setaf 166);
yellow=$(tput setaf 228);
green=$(tput setaf 71);
white=$(tput setaf 15);
bold=$(tput bold);
reset=$(tput sgr0);

PS1="\[${bold}\]\n";
PS1+="\[${orange}\]\u"; # username
PS1+="\[${white}\] @ ";
PS1+="\[${yellow}\]\h";  # host
PS1+="\[${white}\] in ";
PS1+="\[${green}\]\w";   # working directory
PS1+="\n";
PS1+="\[${white}\]>> \[${reset}\]"; # '>>' and reset color
export PS1;

#The below command sets the User name
export USER="SeeBenProgram"
